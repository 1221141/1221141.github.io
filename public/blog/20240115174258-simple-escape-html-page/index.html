Simple escape html page

2024-01-15T17:42:58

When i want to add code to the WordPress posts I need to escape certain characters, I created a page that does that for me:


<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Code Escaper</title>
    <link href="https://stackpath.bootstrapcdn.com/bootstrap/4.3.1/css/bootstrap.min.css" rel="stylesheet">
    <script src="https://code.jquery.com/jquery-3.6.4.min.js"></script>
    <script src="https://stackpath.bootstrapcdn.com/bootstrap/4.3.1/js/bootstrap.min.js"></script>
</head>
<body class="container mt-5">


    <div class="row">
        <div class="col-md-6">
            <label for="inputCode">Input Code:</label>
            <textarea class="form-control" id="inputCode" rows="10"></textarea>
        </div>


        <div class="col-md-6">
            <label for="escapedCode">Escaped Code:</label>
            <textarea class="form-control" id="escapedCode" rows="10" readonly></textarea>
        </div>
    </div>


    <div class="row mt-3">
        <div class="col">
            <button class="btn btn-primary" onclick="escapeCode()">Escape Code</button>
            <button class="btn btn-success" onclick="copyToClipboard()">Copy to Clipboard</button>
        </div>
    </div>


    <script>
        function escapeCode() {
            var inputCode = $('#inputCode').val();
            var escapedCode = $('<div>').text(inputCode).html();
            $('#escapedCode').val(escapedCode);
        }


        function copyToClipboard() {
            $('#escapedCode').select();
            document.execCommand('copy');
            alert('Code copied to clipboard!');
        }
    </script>


</body>
</html>
