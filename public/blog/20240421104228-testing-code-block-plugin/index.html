function exclude_from_loop($query) {
    // Check if it's the main query and not in admin area
    if ($query->is_main_query() && !is_admin()) {
        // Replace 123 with the ID of the tag you want to exclude
        $exclude_tag_id = 102;
        $$exclude_category_id = 105;
        // Exclude posts with the specified tag
        $query->set('tag__not_in', array($exclude_tag_id));
		$query->set('category__not_in', array($exclude_category_id));




    }
}
add_action('pre_get_posts', 'exclude_from_loop');
